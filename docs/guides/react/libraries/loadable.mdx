# Use @loadable in input libraries

```jsx
import Notifiles from '@notifiles/react'
import OvideLocal from 'components/shared/notifiles-library'
import * as Yup from 'yup'
```

```js
import loadable from '@loadable/component'
const keys = [        
    'submit',    
    'input',            
    'textArea',
]

export default (props) => {
    const { type } = props
    if (!keys.includes(type)) {
        return null
    }
    try {
        const Component = loadable(props => import(`./${type}`), {
            cacheKey: props => type,
        })
        return Component
    } catch (e) {
        return null
    }
}



```
